# CSR → SSR Migration Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Convert 7 data pages (dashboard, customers, claims, contracts, proposals, underwriting, admin/users) from CSR to SSR by introducing Server Component shells that prefetch data and hand off to existing Client islands via TanStack Query hydration.

**Architecture:** Each page splits into (a) `page.tsx` Server Component that creates a request-scoped `QueryClient`, prefetches initial data via a new `serverFetch` helper (cookies forwarded via `next/headers`), dehydrates, and (b) `_client.tsx` Client island that hosts the existing page body — `useXxx()` hooks resolve immediately from the hydrated cache. Form pages stay CSR.

**Tech Stack:** Next.js 16 App Router, React 19, TanStack Query 5, TypeScript strict, pnpm.

**Spec:** [docs/superpowers/specs/2026-05-21-csr-to-ssr-design.md](../specs/2026-05-21-csr-to-ssr-design.md)

**Branch:** `feat/csr-to-ssr-data-pages` from `master`.

```ts
import { cookies } from "next/headers";

const BACKEND_URL = process.env.BACKEND_URL ?? "http://localhost:8080";

export async function serverFetch<T>(path: string): Promise<T> {
  const cookieStore = await cookies();
  const cookieHeader = cookieStore
    .getAll()
    .map((c) => `${c.name}=${c.value}`)
    .join("; ");

  const res = await fetch(`${BACKEND_URL}${path}`, {
    headers: { cookie: cookieHeader },
    cache: "no-store",
  });

  if (res.status === 401) {
    const { redirect } = await import("next/navigation");
    redirect("/login");
  }
  if (!res.ok) {
    throw new Error(`Server fetch ${path} failed: ${res.status}`);
  }

  const text = await res.text();
  return (text ? JSON.parse(text) : undefined) as T;
}
```

```tsx
import { QueryClient, dehydrate, HydrationBoundary } from "@tanstack/react-query";
import { serverFetch } from "@/shared/api";
import { customerKeys } from "@/entities/customer";
import type { CustomerDto } from "@/entities/customer";
import type { PageResponse } from "@/shared/api";
import CustomersPageClient from "./_client";

export default async function CustomersPage() {
  const queryClient = new QueryClient();

  await queryClient.prefetchQuery({
    queryKey: customerKeys.list({ page: 0, size: 20 }),
    queryFn: () =>
      serverFetch<PageResponse<CustomerDto>>("/api/customers?page=0&size=20"),
  });

  return (
    <HydrationBoundary state={dehydrate(queryClient)}>
      <CustomersPageClient />
    </HydrationBoundary>
  );
}
```

```tsx
import { healthAnalysisKeys, type HealthAnalysisDto } from "@/entities/health-analysis";

export default async function UnderwritingPage({
  searchParams,

  const queryClient = new QueryClient();

  const prefetches: Promise<unknown>[] = [
```
